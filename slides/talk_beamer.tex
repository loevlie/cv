\begin{frame}{One idea per slide}
Tabular FMs treat \emph{every column} as a token.

\bigskip

The hard part is not the model. It is the \emph{schema}.
\end{frame}

\begin{frame}{Why now?}
\begin{itemize}
  \item Cross-table pretraining: TabPFN, TabuLa-8B, CARTE, GTL
  \item Multimodal grounding: text/image columns alongside numerics
  \item Open data: NeurIPS Table Representation Learning workshop
\end{itemize}
\end{frame}

\begin{frame}{Math}
\[
  p(y \mid x) = \sum_z p(y \mid z)\,p(z \mid x)
\]

\smallskip

{\small\color{meta}Standard latent-variable factorization. Pretraining learns
$p(z\mid x)$ across schemas; downstream fine-tuning conditions on
schema-specific $p(y \mid z)$.}
\end{frame}

\begin{frame}[fragile]{Code, sparingly}
\begin{verbatim}
def encode(row, schema):
    return [
      tokenizer(col_name, value, schema[col_name])
      for col_name, value in row.items()
    ]
\end{verbatim}
\end{frame}

\begin{frame}{Results}
\begin{itemize}
  \item TabPFN-bench: \alert{+6.1} avg accuracy
  \item 14/14 downstream benchmarks improved
  \item Zero per-task fine-tuning
\end{itemize}
\end{frame}

\begin{frame}{Where we go next}
\begin{enumerate}
  \item \emph{Larger schemas} — 1k+ columns
  \item \emph{Image-column pretraining} at scale
  \item \emph{Probabilistic head} — Bayesian uncertainty
\end{enumerate}
\end{frame}
